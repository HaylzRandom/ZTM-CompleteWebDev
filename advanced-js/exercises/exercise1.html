Advanced JavaScript Exercises - Exercise 1

Question: For all of these, what is the value of a when the function gets called with the alert()

Q1

	
function q1() {
 

 var a = 5;
 

 if (a > 1) {
     a = 3;
 }
     alert(a);
}

Answer: The value shown will be 3

Q2

				
var a = 0;


function q2() {
  a = 5;
}

function q22() {
  alert(a);
}

Answer: The value shown will be 5. First run function q2 before q22

Q3

				
function q3() {
  window.a = 'hello';
}

function q32() {
  alert(a);
}

Answer: Running q3 first and then q32 will display 'hello'

Q4

				
var a = 2;
if (true) {
  var a = 5;
  alert(a);
}
alert(a);

Answer: alert(a) will be 5
