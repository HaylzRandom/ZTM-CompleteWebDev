Q1: Create a function called biggestNumberInArray() that takes and array as a parameter and returns the biggest number. Use at least 3 different types of loops.

	
const array = [-1,0,3,100, 99, 2, 99] // should return 100
const array2 = ['a', 3, 4, 2] // should return 4
const array3 = [] // should return 0

Answer: 

							
function biggestNumberInArray(arr) {
	let highestNum = 0;
	for (let i = 0; i < arr.length; i++) {
		if (highestNum < arr[i]) {
			highestNum = arr[i];
		}
	}
	return highestNum;
}

function biggestNumberInArray2(arr) {
	let highestNum = 0;
	arr.forEach((number) => {
		if (highestNum < number) {
			highestNum = number;
		}
	});
	return highestNum;
}

function biggestNumberInArray3(arr) {
	let highestNum = 0;
	for (number of arr) {
		if (highestNum < number) {
			highestNum = number;
		}
	}

	return highestNum;
}

Q2: Write a function checkBasket() that lets you know if the item is in the basket or not

				
amazonBasket = {
  glasses: 1,
  books: 2,
  floss: 100
}

Answer: 

							
function checkBasket(basket, lookingFor) {
	for (item in basket) {
		if (item === lookingFor) {
			return `Item ${lookingFor} has been found!`;
		}
	}
	return `Sorry ${lookingFor} isn't in your basket :(`;
}
