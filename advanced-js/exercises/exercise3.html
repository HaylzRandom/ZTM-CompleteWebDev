Q1

	
// let + const
var a = 'test';
var b = true;
var c = 789;
a = 'test2';

Answer: 

let a = 'test';
let b = true;
const c = 789;
a = 'test2';

Q2

				
// Destructuring
var person = {
    firstName : "John",
    lastName  : "Doe",
    age       : 50,
    eyeColor  : "blue"
};

var firstName = person.firstName;
var lastName = person.lastName;
var age = person.age;
var eyeColor = person.eyeColor;

Answer:
const { firstName, lastName, age, eyeColor } = person;

Q3

var b = true;
var c = 789;

var okObj = {
  a: a,
  b: b,
  c: c
};

Answer:

const b = true;
const c = 789;

    if (username && location) {
        return "I am not lost";
    } else {
        return "I am totally lost!";
    }
}
