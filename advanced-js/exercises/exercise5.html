Q1: Create an array using forEach that has all the usernames with a "!" to each of the usernames

Answer: 

let newUsernames = [];
array.forEach((user) => {
	let { username } = user;
	username + username + '!';
	newUsernames.push(username);
});

console.log('forEach', newUsernames);

Q2: Create an array using map that has all the usernames with a "? to each of the usernames

Answer: 


const mapArray = array.map((user) => {
	let { username } = user;
	return (username += '?');
});

console.log('mapArray', mapArray);

Q3: Filter the array to only include users who are on team: red

Answer: 

                            
const filterArray = array.filter((user) => {
	return user.team === 'red';
});

console.log('filterArray', filterArray);

Q4: Find out the total score of all users using reduce

Answer: 

                            
const reduceArray = array.reduce((acc, user) => {
	return acc + user.score;
}, 0);

Q5: What is the value of i and make this map function pure:

const newArray = arrayNum.map((num, i) => {
	console.log(num, i);
	alert(num);
	return num * 2;

Answer: Value of i is index of array 

                            
const newArray = arrayNum.map((num, i) => {
