Q3: Create two classes: an Animal class and a Mammal class. Create a cow that accepts a name, type and color and has a sound method that moo's her name, type and color.

Answer:

                            
class Animal {
    constructor(name, type, color) {
        this.name = name;
        this.type = type;
        this.color = color;
    }
}

class Mammal extends Animal {
    constructor(name, type, color) {
        super(name, type, color)
    }
    sound() {
        console.log(`Mooo! My name is ${this.name} 
        and I am a ${this.type} and my colourisation 
        is ${this.color}`)
    }
}

const cow = new Mammal('Clarabell', 'cow', 'white');
console.log(cow);
cow.sound();
